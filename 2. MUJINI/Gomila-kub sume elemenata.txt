import pyGomile

class prosirenaGomila(pyGomile.Gomila):
    
    def kubSume(self, n): # Parametar n - veliƒçina gomile
        if(self.velicina==0):
            print("\nGomila je prazna!\n")
        else:            
            suma = int(0)            
            for i in range (0,n):
                suma = suma + int(self.elementi[i])
            
            print("Kub sume elemenata gomile je: " + str(suma*suma*suma))
